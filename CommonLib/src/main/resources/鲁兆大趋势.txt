{鲁兆大趋势}

MA19: MA(CLOSE, 19);
MA43: MA(CLOSE, 43);
MA86: MA(CLOSE, 86);

STICKLINE(MA19<MA43 AND CLOSE>MA19 , MA19,MA43,1,1),COLORWHITE;
STICKLINE(MA19>MA43 AND MA43<MA86 , MA19,MA43,1,1),COLORYELLOW;

{
震出东方: IF(CLOSE>MA19 AND REF(MA19,1)<MA19, 1, 0);
升越良山: IF(CROSS(MA19, MA43) AND CLOSE>MA19 AND CLOSE>MA43, 1, 0);
三山重叠: IF(CROSS(MA43, MA86), 1, 0);
}
