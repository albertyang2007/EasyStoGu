#Keyspace
CREATE KEYSPACE IF NOT EXISTS EasyStoGu WITH replication = {'class':'SimpleStrategy', 'replication_factor':1};

#ind_macd table
CREATE TABLE IF NOT EXISTS EasyStoGu.ind_macd
(
  stockid text,
  date text,
  dif double,
  dea double,
  macd double,
  PRIMARY KEY (stockid, date)
) ;

#ind_macd_index (has some problem!!!)
CREATE CUSTOM INDEX ind_macd_index ON EasyStoGu.ind_macd ()
USING 'com.stratio.cassandra.lucene.Index'
WITH OPTIONS = {
   'refresh_seconds': '10',
   'schema': '{
      fields: {
         stockid: {type: "string"},
         date: {type: "string"}
      }
   }'
};
